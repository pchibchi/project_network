  !SEOUL_GW1

  !SEOUL_GW2

  !for SPOKE-HUB, static
  match identity address 14.14.14.2 255.255.255.255
  match identity address 14.14.14.6 255.255.255.255
  !for SPOKE-SPOKE, dynamic 
  match identity address 0.0.0.0

  local-address 26.26.26.2

!

  set transform-set VPN
  set security-association lifetime seconds 3600
